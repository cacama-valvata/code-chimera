    import urllib
    import mimetypes
    
    
    def guess_type_of(link, strict=True):
        link_type, _ = mimetypes.guess_type(link)
        if link_type is None and strict:
            u = urllib.urlopen(link)
            link_type = u.headers.gettype() # or using: u.info().gettype()
        return link_type
